// Funzioni di test
function testFesta(tipo) {
    const body = document.body;
    const dashboard = document.getElementById('dashboard');
    const auguriElement = document.getElementById('auguriSpeciali');
    
    // Rimuovi tutte le classi
    const classiFesta = [
        'festa-natale', 'festa-halloween', 'festa-pasqua', 
        'festa-san-valentino', 'festa-memoria', 'festa-donna',
        'festa-papa', 'festa-nonni', 'festa-mamma',
        'festa-gaudenzio', 'festa-repubblica'
    ];
    
    classiFesta.forEach(classe => {
        body.classList.remove(classe);
        dashboard.classList.remove(classe);
    });
    
    if (tipo) {
        body.classList.add('festa-' + tipo);
        dashboard.classList.add('festa-' + tipo);
        
        // Messaggi di test
        const messaggi = {
            'natale': '🎄 TEST - Buon Natale e Felice Anno Nuovo! 🎄',
            'halloween': '🎃 TEST - Dolcetto o scherzetto? Buon Halloween! 🎃',
            'pasqua': '🐣 TEST - Buona Pasqua! 🐣',
            'san-valentino': '❤️ TEST - Buon San Valentino! ❤️',
            'memoria': '🕯️ TEST - Giornata della Memoria · Per non dimenticare',
            'donna': '🌸 TEST - Festa della Donna · 8 Marzo',
            'papa': '👔 TEST - Buona Festa del Papà!',
            'nonni': '👴👵 TEST - Buona Festa dei Nonni!',
            'mamma': '🌷 TEST - Buona Festa della Mamma!',
            'gaudenzio': '🎉 TEST - Buon San Gaudenzio, patrono di Rimini!',
            'repubblica': '🇮🇹 TEST - Festa della Repubblica Italiana!'
        };
        
        auguriElement.textContent = messaggi[tipo];
        auguriElement.style.display = 'block';
    } else {
        auguriElement.style.display = 'none';
    }
}
